### 2.1 缓存层建设

**目标**：缓存命中率 > 40%，降低 30-60% API 调用成本

#### 2.1.1 精确缓存（Exact Match）

相同 prompt + 参数直接返回缓存结果：

```python
# backend/app/core/cache/exact_cache.py
import hashlib
import json
from redis import Redis
from functools import wraps

redis_client = Redis(host="redis", decode_responses=True)

def cache_key(prompt: str, params: dict) -> str:
    """生成缓存 key：prompt 哈希 + 参数哈希"""
    content = json.dumps({"prompt": prompt, **params}, sort_keys=True)
    return f"cache:exact:{hashlib.sha256(content.encode()).hexdigest()[:16]}"

async def get_cached_result(cache_key: str) -> str:
    """返回缓存的生成物 URL"""
    return redis_client.get(cache_key)

async def set_cached_result(cache_key: str, url: str, ttl: int = 86400):
    """缓存结果，默认 24 小时"""
    redis_client.setex(cache_key, ttl, url)
```

**缓存命中条件**：相同用户 + 相同 prompt + 相同参数 + 短时间内（可配置）

## 五、积分系统增强

### 5.1 三重幂等性保障完善

增长阶段应实现完整的幂等性保障：

```python
# backend/app/core/points/service.py

class PointsService:
    async def deduct_with_triple_idempotency(
        self,
        user_id: str,
        amount: int,
        transaction_id: str,
        idempotency_key: str
    ) -> bool:
        """
        三重幂等性保障：
        1. 幂等键（数据库唯一索引）
        2. 分布式锁（Redis）
        3. 乐观锁（版本号）
        """
        # 第一重：分布式锁
        lock_key = f"lock:points:deduct:{user_id}"
        lock = redis.lock(lock_key, timeout=30, blocking_timeout=30)
        
        if not await lock.acquire():
            raise PointsError("系统繁忙，请稍后重试")
        
        try:
            async with self.db.transaction():
                # 第二重：幂等键检查
                existing = await self.db.fetch_one(
                    """
                    SELECT id FROM points_transactions 
                    WHERE idempotency_key = :key
                    """,
                    {"key": idempotency_key}
                )
                if existing:
                    return False  # 已处理
                
                # 第三重：乐观锁更新
                result = await self.db.execute(
                    """
                    UPDATE points_accounts 
                    SET balance = balance - :amount,
                        version = version + 1
                    WHERE user_id = :user_id 
                      AND balance >= :amount
                      AND version = :expected_version
                    """,
                    {
                        "amount": amount,
                        "user_id": user_id,
                        "expected_version": await self.get_version(user_id)
                    }
                )
                
                if result.rowcount == 0:
                    raise InsufficientBalance()
                
                # 记录流水
                await self.db.execute(
                    """
                    INSERT INTO points_transactions 
                    (user_id, transaction_id, idempotency_key, type, amount)
                    VALUES (:user_id, :tid, :key, 'SPEND', :amount)
                    """,
                    {"user_id": user_id, "tid": transaction_id, 
                     "key": idempotency_key, "amount": amount}
                )
                
                return True
        finally:
            await lock.release()
```

### 5.2 预扣费机制完善

```python
# 任务开始：预扣最大可能积分
async def task_start(user_id: str, task_id: str, model: str):
    max_cost = get_max_cost(model)  # 查询模型最大消耗
    
    # 冻结积分
    await points_service.freeze(
        user_id=user_id,
        amount=max_cost,
        transaction_id=f"FREEZE:{task_id}",
        reference_id=task_id,
        reference_type="task"
    )
    
    # 记录任务状态
    await task_service.update(task_id, {
        "status": "processing",
        "frozen_points": max_cost
    })

# 任务完成：多退少补
async def task_complete(task_id: str, actual_cost: int):
    task = await task_service.get(task_id)
    frozen = task.frozen_points
    
    if actual_cost < frozen:
        # 退还多余部分
        await points_service.unfreeze(user_id, frozen - actual_cost, task_id)
    elif actual_cost > frozen:
        # 扣除额外部分
        await points_service.deduct(user_id, actual_cost - frozen, task_id)
    
    await task_service.update(task_id, {
        "status": "completed",
        "actual_cost": actual_cost
    })
```

# backend/app/core/fraud_detection/service.py

class FraudDetectionService:
    def __init__(self, redis: Redis):
        self.redis = redis
    
    async def check_sign_abuse(self, user_id: str) -> bool:
        """检测签到作弊"""
        # 检查同一 IP 多账号签到
        ip = await self.get_user_ip(user_id)
        ip_sign_count = await self.redis.scard(f"sign:ip:{ip}")
        
        if ip_sign_count > 3:
            return True  # 拒绝
        
        # 检查异常签到时间（凌晨批量签到）
        sign_hour = datetime.now().hour
        if sign_hour < 5:
            return True
        
        return False
    
    async def check_invite_abuse(self, inviter_id: str) -> bool:
        """检测邀请作弊"""
        # 检查邀请频率
        invites_today = await self.redis.get(f"invite:{inviter_id}:today")
        if int(invites_today or 0) > 10:
            return True
        
        # 检查被邀请者的设备指纹
        device_fingerprints = await self.redis.smembers(
            f"invite:{inviter_id}:devices"
        )
        if len(device_fingerprints) < 2:
            return True  # 几乎无独立设备
        
        return False
    
    async def rate_limit_api(self, user_id: str, endpoint: str) -> bool:
        """API 限流（令牌桶）"""
        key = f"ratelimit:{endpoint}:{user_id}"
        
        count = await self.redis.incr(key)
        if count == 1:
            await self.redis.expire(key, 60)  # 1 分钟窗口
        
        limit = self.get_endpoint_limit(endpoint)
        return count <= limit
